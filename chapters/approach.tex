\section{ต้นแบบจำลองโมเดล (Prototype Model for Two-Player Case)}
ในการทดลองนี้ ได้เริ่มจากการพัฒนาต้นแบบจำลองสำหรับกรณีผู้เล่น 2 คน เพื่อใช้เป็นกรณีศึกษาพื้นฐานในการทดสอบความถูกต้องของแนวทางการสร้างแบบจำลองเกมเชิงกลยุทธ์และการแปลงปัญหาให้อยู่ในรูปแบบของ Quadratic Unconstrained Binary Optimization (QUBO) โดยมีวัตถุประสงค์เพื่อวิเคราะห์ความสามารถของแบบจำลองในการหาดุลยภาพแนช (Nash Equilibrium) และตรวจสอบความเหมาะสมก่อนนำไปขยายผลในกรณีที่มีความซับซ้อนมากขึ้น

% --- 2 player part 1 ---

\subsection{กรณีผู้เล่น 2 คน (ตารางขนาด 2×2)}
กำหนดให้มีผู้เล่น 2 คน ได้แก่ ผู้เล่น A และผู้เล่น B โดยแต่ละผู้เล่นมีทางเลือกเชิงกลยุทธ์จำนวน 2 ทางเลือก คือ ไม่รับสารภาพ (0) และ รับสารภาพ (1) สามารถแสดงผลตอบแทน (Payoff Matrix) ได้ดังนี้

\begin{table}[h]
\centering
\begin{tabular}{|c|c|c|}
\hline
\textbf{A \textbackslash B} & \textbf{0 (ไม่สารภาพ)} & \textbf{1 (สารภาพ)} \\ \hline
\textbf{0 (ไม่สารภาพ)} & (1,1) & (0,4) \\ \hline
\textbf{1 (สารภาพ)} & (4,0) & (3,3) \\ \hline
\end{tabular}
\caption{Payoff Matrix ขนาด 2×2 ของ Prisoner's Dilemma}
\end{table}

\paragraph{วิธีการหาดุลยภาพแนช (Nash Equilibrium)}
ดุลยภาพแนชคือสถานะที่ไม่มีผู้เล่นคนใดสามารถเพิ่มผลตอบแทนของตนเองได้โดยการเปลี่ยนกลยุทธ์เพียงฝ่ายเดียว จากการวิเคราะห์ความพึงพอใจสูงสุด (Best Response) ของทั้งสองผู่เล่นในการจำลอง พบว่าสถานะ $(x,y) = (1,1)$ เป็นดุลยภาพแนชของเกมนี้ ซึ่งในมุมมองของ QUBO จะสอดคล้องกับสถานะที่ทำให้ฟังก์ชันพลังงาน (Energy level) ที่มีค่าน้อยที่สุด

\paragraph{การสร้างแบบจำลองในรูปแบบ QUBO}
กำหนดตัวแปรไบนารี $x, y \in \{0,1\}$ แทนกลยุทธ์ของผู้เล่น A และ B ตามลำดับ โดยใช้หลักการของ Indicator Function ในการสร้างฟังก์ชันผลตอบแทน $f(x,y)$ จะได้ว่า
\begin{equation}
f(x,y) = 1(1-x)(1-y) + 4(1-x)y + 0x(1-y) + 3xy
\end{equation}

\noindent เพื่อให้แบบจำลองโน้มเอียงไปยังสถานะที่ต้องการ จึงมีการเพิ่มฟังก์ชันบทลงโทษ (Penalty Function) 
\begin{equation}
g(x,y) = f(x,y) + \alpha(1-x) + \beta y
\end{equation}
โดยกำหนดให้ $\alpha=1, \beta=1$ เพื่อนำฟังก์ชันไปแก้ด้วย Quantum Annealing ได้

% --- 2 player part 2 ---
\subsection{กรณีผู้เล่น 2 คน (ตารางขนาด n×m)}
เพื่อขยายความสามารถให้รองรับสถานการณ์ความเป็นจริงจึงได้ใช้ข้อมูลของบริษัท Aid-gent ซึ่งได้เรียนรู้มาจากรายวิชา 261405 เพื่อนำมาใช้เป็นสถานการณ์ตัวอย่าง โดยผู้จัดทำได้จำลองระบบเครือข่ายเป็นโครงสร้างต้นไม้โดยมีรายละเอียดดังนี้

\paragraph{การสร้างแบบจำลองโครงสร้างต้นไม้ (Tree-Based Modeling)}
\begin{itemize}
    \item \textbf{การประเมินความเสี่ยง (Risk Assessment)} คำนวณมูลค่าโหนดหรือราคาโหนด (Node's Price, $P$) จากผลคูณระหว่างโอกาสในการเกิดเหตุ (Likelihood) และผลกระทบ (Impact) โดยให้คะแนนอยู่ในช่วง 1--5 และคำนวณต้นทุนขอบ (Edge's Cost, $C$) จากส่วนต่างระหว่างราคาโหนดกับคะแนนการจัดการความเสี่ยง (Treatment Score) ตามมาตรฐานสากล ISO/IEC 27001
    \item \textbf{โครงสร้างลำดับชั้นของต้นไม้ (Tree-Based Hierarchical Structure)} กำหนดโหนดราก (Root Node) ให้เป็นจุดเริ่มต้นของการโจมตี (Initial Attack Point) โดยมีมูลค่าสินทรัพย์ ($P=0$) และจัดวางสินทรัพย์สำคัญ (Critical Assets) ไว้ที่โหนดใบ (Leaf Nodes) เพื่อจำลองกลไกการป้องกันเชิงลึก (Defense in Depth)
    \item \textbf{กลยุทธ์การดำเนินงาน (Strategic Execution)} ฝ่ายโจมตี (Attacker) จะเลือกเส้นทางผ่านอัลกอริทึมการค้นหาตามแนวกว้าง (Breadth-First Search BFS) และการค้นหาตามแนวลึก (Depth-First Search DFS) ในขณะที่ฝ่ายป้องกัน (Defender) จะใช้กลยุทธ์การจัดลำดับความสำคัญของสินทรัพย์แบบจากน้อยไปมาก (Ascending ASC), จากมากไปน้อย (Descending DSC) และแบบสุ่ม (Random)
\end{itemize}

\paragraph{การสร้างแบบจำลองในรูปแบบ QUBO (QUBO Formulation)}
ในการเปลี่ยนปัญหาการตัดสินใจให้อยู่ในรูปแบบสมการคณิตศาสตร์ จะต้องมีการกำหนดเงื่อนไขบังคับ (Constraints) เพื่อให้ผลลัพธ์ที่ได้มีความสมเหตุสมผลตามหลักทฤษฎีเกม ดังนี้

\begin{itemize}
    \item \textbf{การบังคับเลือกกลยุทธ์เดียว (Single Strategy Constraint)} ใช้เทคนิคการเข้ารหัสแบบหนึ่งในหลาย (One-hot Encoding) เพื่อบังคับให้ผู้เล่นแต่ละฝ่าย (ทั้งผู้โจมตีและผู้ป้องกัน) สามารถตัดสินใจเลือกใช้กลยุทธ์ได้เพียง 1 รูปแบบเท่านั้น ต่อการดำเนินงานใน 1 รอบการจำลอง โดยเขียนในรูปสมการผลรวมของตัวแปรการตัดสินใจ ($x_i$ และ $y_j$) จะต้องเท่ากับ 1 เสมอ
    \begin{equation}
    \sum_{i=1}^{n} x_i = 1, \quad \sum_{j=1}^{m} y_j = 1
    \end{equation}

    \item \textbf{ฟังก์ชันเป้าหมายและฟังก์ชันบทลงโทษ (Objective Function with Penalty Terms)} เพื่อให้คอมพิวเตอร์ควอนตัมค้นหาคำตอบที่สอดคล้องกับเงื่อนไขข้างต้น จึงมีการรวมค่าปรับหรือบทลงโทษ (Penalty) เข้าไปในฟังก์ชันพลังงานรวม $H(x,y)$ ดังนี้
    \begin{equation}
    H(x,y) = \sum_{i=1}^{n} \sum_{j=1}^{m} P_{ij} x_i y_j + A\left(\sum_{i=1}^{n} x_i - 1\right)^2 + B\left(\sum_{j=1}^{m} y_j - 1\right)^2
    \end{equation}
    โดยที่
    \begin{itemize}
        \item $\sum \sum P_{ij} x_i y_j$ คือส่วนของ \textbf{ฟังก์ชันเป้าหมาย (Objective Function)} ที่คำนวณจากค่าผลตอบแทน (Payoff) ของแต่ละสถานะกลยุทธ์
        \item $A$ และ $B$ คือ \textbf{ค่าสัมประสิทธิ์บทลงโทษ (Penalty Coefficients)} ซึ่งเป็นค่าคงที่บวกขนาดใหญ่ (Large Positive Constants) 
        \item พจน์กำลังสอง $(...-1)^2$ คือ \textbf{ฟังก์ชันบทลงโทษ (Penalty Function)} ที่จะให้ค่าพลังงานสูงขึ้นอย่างมากหากผู้เล่นเลือกกลยุทธ์ไม่ครบหรือเลือกมากกว่า 1 กลยุทธ์ ซึ่งกลไกของ Quantum Annealing จะพยายามหลีกเลี่ยงสถานะที่มีพลังงานสูงเหล่านี้ และโน้มเอียงไปหาคำตอบที่ทำให้พจน์บทลงโทษมีค่าเป็นศูนย์ (สอดคล้องกับเงื่อนไขบังคับ)
    \end{itemize}
\end{itemize}

% --- 2 player part 3 ---
\paragraph{การวิเคราะห์ดุลยภาพแนช (Nash Equilibrium Analysis)}
ในการระบุว่าสถานะใดเป็นดุลยภาพแนช ระบบจะพิจารณาจากค่าความเสียใจ (Regret, $R$) ของผู้เล่นแต่ละฝ่าย โดยมีรายละเอียดของตัวแปรในสมการพลังงาน $H_{nash}$ ดังนี้

\begin{equation}
H_{nash} = (W \cdot \mathbb{I}[R_D > 0] + W \cdot \mathbb{I}[R_A > 0]) x_i y_j
\end{equation}

\noindent \textbf{คำอธิบายตัวแปร}
\begin{itemize}
    \item $R_D$ และ $R_A$ คือ \textbf{ค่าความเสียใจ (Regret)} ของผู้ป้องกัน (Defender) และผู้โจมตี (Attacker) ตามลำดับ โดยคำนวณจากส่วนต่างระหว่าง "ผลตอบแทนสูงสุดที่เป็นไปได้หากเปลี่ยนกลยุทธ์" กับ "ผลตอบแทนในสถานะปัจจุบัน" หากค่า $R > 0$ แสดงว่าผู้เล่นรายนั้นยังสามารถเลือกกลยุทธ์อื่นที่ให้ผลตอบแทนดีกว่าเดิมได้ (ไม่ใช่จุดสมดุล)
    \item $\mathbb{I}[\cdot]$ คือ \textbf{ฟังก์ชันตัวชี้วัด (Indicator Function)} ซึ่งมีสมบัติทางตรรกศาสตร์คือ จะมีค่าเป็น 1 เมื่อเงื่อนไขในวงเล็บเป็นจริง ($R > 0$) และมีค่าเป็น 0 เมื่อเงื่อนไขเป็นเท็จ ($R \leq 0$) 
    \item $W$ คือ \textbf{ค่าน้ำหนักบทลงโทษ (Penalty Weight)} ซึ่งเป็นค่าคงที่บวกขนาดใหญ่ที่ใช้เพื่อเพิ่มระดับพลังงานให้กับสถานะที่ไม่ใช่ดุลยภาพแนช เพื่อให้ระบบ Quantum Annealing หลีกเลี่ยงคำตอบเหล่านั้น
    \item $x_i$ และ $y_j$ คือ \textbf{ตัวแปรการตัดสินใจ (Decision Variables)} ของผู้เล่นแต่ละฝ่ายในสถานะที่กำลังพิจารณา
    \item $H_{nash}$ คือ \textbf{ฟังก์ชันพลังงานรวม (Total Energy Function)} ซึ่งจะมีค่าต่ำสุดเท่ากับ 0 (Zero Energy State) ก็ต่อเมื่อทั้งผู้โจมตีและผู้ป้องกันไม่มีความเสียใจเหลืออยู่ ($R_D = 0$ และ $R_A = 0$) ซึ่งสอดคล้องกับนิยามของดุลยภาพแนชในทางทฤษฎีเกม
\end{itemize}

\section{โมเดลจำลองจริงสำหรับผู้เล่น N คน (General N-Player Model)}
หลังจากสร้างโมเดลต้นเเบบเพื่อทดสอบการคำนวณทางคณิตศาสตร์ และโค้ดได้อย่างถูกต้องแล้ว จึงได้มีการขยายขอบเขตสู่กรณีผู้เล่น $N$ คน เพื่อรองรับสถานการณ์ที่มีผู้โจมตีหลายราย (Multi-attackers) เพื่อให้สอดคล้องกับสถานการณ์ความเป็นจริงมากขึ้น โดยมีขั้นตอนดังนี้

% --- 3.2.1 Risk and Value Assessment ---
\subsection{การวิเคราะห์ความเสี่ยงและการประเมินมูลค่าสินทรัพย์ (Risk and Value Assessment)}
เพื่อให้แบบจำลองมีความสอดคล้องกับสถาปัตยกรรมความปลอดภัยของบริษัทในความเป็นจริง ผู้จัดทำจึงได้ใช้ข้อมูลสินทรัพย์ของบริษัท Aid-gent ที่ได้เรียนรู้ในรหัสวิชา 261405 มาประยุกต์ใช้ โดยกำหนดให้วิธีการคำนวณมูลค่าโหนด ($P$) และต้นทุนขอบ ($C$) โดยอ้างอิงตามมาตรฐาน ISO/IEC 27001 ดังนี้

%  --- คำนวนโหนด ---
\subsubsection{การคำนวณมูลค่าโหนด (Node's Price $P$)}
มูลค่าของสินทรัพย์หรือโหนด ($P$) คำนวณจากคะแนนความเสี่ยง (Risk Score) ซึ่งเป็นผลคูณระหว่างระดับความน่าจะเป็น (Likelihood) และระดับผลกระทบ (Impact)
\begin{equation}
P = \text{Likelihood (1--5)} \times \text{Impact (1--5)}
\end{equation}
โดยเกณฑ์การวัดระดับความรุนแรง (Risk Rating Scale) แบ่งออกเป็น 4 ระดับ คือ ต่ำ (1--4), ปานกลาง (5--9), สูง (10--15) และสูงมาก (16--25)

%  --- คำนวนเส้น หรือ cost ---
\subsubsection{การคำนวณต้นทุนขอบ (Edge's Cost $C$)}
ต้นทุนที่ผู้โจมตีต้องใช้ในการผ่านด่านการป้องกัน ($C$) คำนวณจากส่วนต่างระหว่างมูลค่าโหนดเดิมกับคะแนนการจัดการความเสี่ยง (Treatment Score)
\begin{equation}
C = P - \text{Treatment Score}
\end{equation}
โดยที่ Treatment Score คือคะแนนความเสี่ยงที่คาดว่าจะลดลงหลังจากประยุกต์ใช้มาตรการควบคุมตามมาตรฐานความปลอดภัย

\begin{table}[h]
\centering
\label{tabrisk_calc}
\small
\begin{tabular}{|l|c|c|c|}
\hline
\textbf{สินทรัพย์ (Asset)} & \textbf{Risk Score ($P$)} & \textbf{Treatment Score} & \textbf{Edge Score ($C$)} \\ \hline
Patient Health Data (PHD) & 25 & 2 & 23 \\ \hline
Customer Database (RDS)   & 25 & 2 & 23 \\ \hline
Core Platform             & 15 & 2 & 13 \\ \hline
Security Software         & 15 & 3 & 12 \\ \hline
VDO Call System           & 9  & 4 & 5  \\ \hline
\end{tabular}
\caption{ตัวอย่างการคำนวณค่า $P$ และ $C$ ของสินทรัพย์สำคัญในระบบ Aid-gent}
\end{table}

% --- 3.2.2 อธิบายตาราง payoff  เเละกลยุทธ์ ---
\subsection{การสร้างโครงสร้างต้นไม้ลําดับชั้น (Hierarchical Tree Structure Construction)}
ผู้จัดทำได้ทำการประยุกต์ใช้ทฤษฎีกราฟ (Graph Theory) ในการจำลองสถาปัตยกรรมของระบบรักษาความปลอดภัยแบบหลายชั้น (Defense in Depth)  โดยมีรายละเอียดดังนี้
\begin{itemize}
\item \textbf{โหนด (Nodes)} แทนทรัพยากรหรือสินทรัพย์ที่มีมูลค่า (Valuable Resources) โดยแต่ละโหนดจะมีค่า $P$ (Price) แทนมูลค่าของสินทรัพย์นั้น ๆ 
\item \textbf{เส้นเชื่อม (Edges)} แทนเส้นทางการบุกรุกหรือความเชื่อมโยงระหว่างระบบ โดยมีค่า $C$ (Cost) แทนต้นทุนที่ผู้โจมตีต้องใช้ในการผ่านด่านการป้องกัน 
\item \textbf{โครงสร้างลําดับชั้น} กำหนดให้มีโหนดราก (Root) เป็นจุดเริ่มต้น และกระจายไปยังโหนดใบ (Leaves) ซึ่งเป็นสินทรัพย์ชั้นในสุด 
\end{itemize}

\subsection{การคำนวณและแปลงข้อมูลเป็นตารางผลตอบแทน (Payoff Table Generation)}
หลังจากดำเนินการสร้างโครงสร้างต้นไม้ (Tree Structure) เรียบร้อยแล้ว ขั้นตอนต่อไปคือการใช้กลยุทธ์การสำรวจกราฟ (Graph Traversal Strategies) เพื่อคำนวณค่าผลตอบแทน (Payoff) ที่ผู้เล่นแต่ละฝ่ายจะได้รับในสถานการณ์จำลองต่าง ๆ ดังนี้

\begin{itemize}
    \item \textbf{กลยุทธ์ฝ่ายป้องกัน (Defender Strategies)} เช่น การจัดลำดับความสำคัญของสินทรัพย์ (Asset Prioritization) ในรูปแบบจากน้อยไปมาก (Ascending ASC), จากมากไปน้อย (Descending DSC) หรือแบบสุ่ม (Random) เพื่อกำหนดแนวทางการวางมาตรการป้องกันระบบ
    \item \textbf{กลยุทธ์ฝ่ายโจมตี (Attacker Strategies)} ใช้ระเบียบวิธีค้นหาตามแนวกว้าง (Breadth-First Search BFS) และการค้นหาตามแนวลึก (Depth-First Search DFS) เพื่อหาเส้นทางการบุกรุกภายใต้ข้อจำกัดด้านงบประมาณ (Budget Constraints)
    \item \textbf{การบันทึกค่าผลตอบแทน (Payoff Recording)} ระบบจะประมวลผลการบุกรุกในรูปแบบการสลับตากันเล่น (Turn-based) และนำมูลค่าของโหนด (Node Value) ที่ถูกโจมตีได้สำเร็จมาคำนวณเป็นผลตอบแทนของผู้โจมตี ในขณะที่มูลค่าของสินทรัพย์ที่เหลือรอดจากการโจมตีจะถูกนับเป็นผลตอบแทนของผู้ป้องกัน โดยข้อมูลทั้งหมดจะถูกจัดเก็บในรูปแบบตารางผลตอบแทน (Payoff Matrix) เพื่อใช้เป็นข้อมูลนำเข้า (Input Data) สำหรับแบบจำลอง QUBO ในขั้นตอนถัดไป
\end{itemize}


% เพิ่มรูปตารางที่ได้หลังจากทำแผนภาพต้นไม้
\paragraph{การจำลองสถานการณ์และตารางผลตอบแทน (Simulation and Payoff Table)}
จากการนำโครงสร้างต้นไม้ที่พัฒนาขึ้นมาทดสอบในสถานการณ์จำลอง (Simulation Scenario) โดยกำหนดให้ฝ่ายป้องกัน (Defender) เลือกใช้กลยุทธ์การจัดลำดับความสำคัญของสินทรัพย์ (Asset Prioritization) ทั้ง 3 รูปแบบ ได้แก่

\subsection{ระเบียบวิธีและขั้นตอนการประมวลผลเชิงคำนวณ (Computational Methodology)}
เพื่อให้แบบจำลองสามารถระบุจุดดุลยภาพแนชได้อย่างมีประสิทธิภาพภายใต้ข้อจำกัดของทรัพยากรควอนตัมในปัจจุบัน โครงงานนี้จึงนำเสนอขั้นตอนการประมวลผลแบบผสมผสาน (Hybrid Approach) โดยมีรายละเอียดตามลำดับขั้นตอนดังนี้

\subsubsection{การกำหนดตัวแปรและเงื่อนไขการเลือก (Variables and Selection Constraints)}
กำหนดให้ $s_j^i$ เป็นตัวแปรไบนารี (Binary Variable) แทนการตัดสินใจของผู้เล่น $p_i$ ในการเลือกกลยุทธ์ $j$ โดยมีนิยามคือ
\begin{equation}
s_j^i = \begin{cases} 1 & \text{หากกลยุทธ์ } j \text{ ของผู้เล่น } p_i \text{ ถูกเลือกใช้งาน} \\ 0 & \text{ในกรณีอื่น ๆ} \end{cases}
\end{equation}

เพื่อให้สอดคล้องกับหลักทฤษฎีเกมและการตรวจสอบสถานะกลยุทธ์ในรูปแบบ QUBO ผู้เล่นแต่ละรายจะถูกกำหนดเงื่อนไขบทลงโทษ (Constraint) ให้อยู่ในรูปฟังก์ชันกำลังสองเพื่อให้มีค่าพลังงานต่ำสุดเป็นศูนย์เมื่อมีการเลือกกลยุทธ์ที่ถูกต้อง
\begin{equation}
\left( \sum_{j=1}^{m^i} s_j^i - m^i \right)^2 = 0
\end{equation}
\noindent \textbf{คำอธิบายตัวแปร} $m^i$ คือจำนวนกลยุทธ์ทั้งหมดของผู้เล่น $i$ ซึ่งการกำหนดให้ผลรวมเท่ากับ $m^i$ ช่วยให้ระบบสามารถตรวจสอบสถานะของทุกกลยุทธ์ได้พร้อมกันในการประมวลผลหนึ่งหน่วยย่อย

\subsubsection{การสร้างสมการ QUBO สำหรับดุลยภาพแนช (QUBO Formulation)}
เพื่อให้เงื่อนไขอสมการดุลยภาพแนช $v_j^i - v_k^i - 1 \geq 0$ สามารถคำนวณได้บน Quantum Annealer จะต้องแปลงให้อยู่ในรูปสมการกำลังสองที่เท่ากับ 0 โดยใช้ตัวแปรเสริม (Slack Variable $s_k$) เข้ามาช่วย โดยสมการพลังงานรวมสำหรับผู้เล่น $p_i$ ในกลยุทธ์ $s_j^i$ (กำหนดให้เป็น $\phi_i$) สามารถเขียนได้ดังนี้

\begin{equation}
\begin{aligned}
\phi_i = & \left( s_1^i + s_2^i + s_3^i + \dots + s_{m_i}^i - m^i \right)^2 + \\
& \left( v_j^i s_j^i - v_{k_1}^i s_{k_1}^i - 1 - s_1 \right)^2 + \dots + \\
& \left( v_j^i s_j^i - v_{k_{m_i}}^i s_{k_{m_i}}^i - 1 - s_{m_i} \right)^2
\end{aligned}
\end{equation}

\noindent โดยที่ $\phi_i$ คือผลลัพธ์ของคำตอบสำหรับผู้เล่น $p_i$ เมื่อเลือกกลยุทธ์ $s_j^i$ ซึ่งมีเงื่อนไขสำคัญคือ \textbf{หาก $\phi_i = 0$ แสดงว่าค่านี้คือดุลยภาพแนช (NASH)} เนื่องจากทุกเงื่อนไขในพจน์กำลังสองถูกเติมเต็มทั้งหมด

\subsubsection{การเข้ารหัสตัวแปรเสริมด้วย Binary Expansion}
ตัวแปรเสริม $s_k$ จะถูกแทนด้วยคิวบิตผ่านการขยายเชิงเลขฐานสอง (Binary Expansion) เพื่อหาค่าจำนวนเต็มที่เหมาะสมที่สุดในระหว่างการประมวลผล
\begin{equation}
s_k = \sum_{n=0}^{N-1} 2^n q_n = 1q_0 + 2q_1 + 4q_2 + 8q_3 + 16q_4 + \dots
\end{equation}
\noindent \textbf{คำอธิบายตัวแปร}
\begin{itemize}
    \item $q_n \in \{0, 1\}$ คือคิวบิตที่ Quantum Annealer ใช้ในการค้นหาค่าที่ทำให้สมการสมบูรณ์
    \item $2^n$ คือค่าน้ำหนัก (Weight) สำหรับแต่ละบิตเพื่อสร้างค่าจำนวนเต็ม
    \item $N$ ในโครงงานนี้เลือกใช้ $N=5$ ซึ่งรองรับส่วนต่างของผลตอบแทน (Payoff Difference) ได้สูงสุด $2^5 - 1 = 31$
\end{itemize}

\subsubsection{แนวคิดไฮบริดและการบริหารทรัพยากรคิวบิต (Hybrid Quantum-Classical Approach)}
เนื่องจากการรวมทุกเงื่อนไขเข้าเป็นสมการเดียว (Total QUBO) จะทำให้เกิดพจน์ที่ไม่ใช่กำลังสอง (Non-Quadratic) ซึ่งต้องใช้ทรัพยากรคิวบิตมหาศาลเกินขีดจำกัดของฮาร์ดแวร์ในปัจจุบัน เราจึงเลือกใช้แนวทางไฮบริดดังนี้
\begin{enumerate}
    \item \textbf{Classical Orchestration} คอมพิวเตอร์ดั้งเดิมทำหน้าที่จัดการลูปการทำงาน (Iterative loops) เพื่อส่งข้อมูลในแต่ละโปรไฟล์กลยุทธ์ (Strategy Profile) เข้าสู่ระบบ
    \item \textbf{Isolated QUBO Cell} ส่งเฉพาะสมการ $\phi_i$ ของแต่ละเซลล์ที่ต้องการตรวจสอบให้ Quantum Annealer ประมวลผลเป็นหน่วยย่อย (Cell-by-cell)
    \item \textbf{Qubit Reuse} เมื่อคำนวณเสร็จในแต่ละหน่วย ระบบจะทำการ Reset และคืนทรัพยากรคิวบิตเพื่อนำกลับมาใช้ใหม่ในรอบถัดไป
\end{enumerate}

\subsubsection{การพิสูจน์ความสอดคล้องของแบบจำลอง (Backward \& Forward Proof)}
\begin{itemize}
    \item \textbf{Forward Proof ($NE \rightarrow \phi_i = 0$)} หากกลยุทธ์เป็นดุลยภาพแนช ผลตอบแทนปัจจุบันต้องมากกว่าทางเลือกอื่นเสมอ ทำให้ค่าในวงเล็บเป็นจำนวนเต็มบวก ซึ่งระบบสามารถเลือกค่า Slack Variable ที่เหมาะสมเพื่อทำให้พจน์กำลังสองมีค่าเป็นศูนย์ ส่งผลให้ $\phi_i = 0$
    \item \textbf{Backward Proof ($\phi_i = 0 \rightarrow NE$)} หากตรวจพบพลังงานต่ำสุดเป็นศูนย์ เนื่องจากสมการอยู่ในรูปผลรวมกำลังสอง ทุกพจน์ย่อยต้องเป็นศูนย์เท่านั้น นั่นคือ $v_j^i s_j^i - v_k^i s_k^i - 1 = s_k$ และเมื่อ $s_k \geq 0$ จึงพิสูจน์ได้ว่าผลตอบแทนปัจจุบันดีกว่าทางเลือกอื่นจริง
\end{itemize}

\begin{algorithm}[H]
\caption{Hybrid Nash Verification with Qubit Management}
\begin{algorithmic}[1]
\FOR{each \textit{strategy\_profile} \textbf{in} \textit{all\_possible\_profiles}}
    \STATE $H_{cell} \gets 0$
    \STATE \textbf{Allocate} qubits for \textit{slack\_variables}
    \FOR{each \textit{player} $i$}
        \STATE $v_{curr} \gets \text{get\_payoff}(i, \textit{strategy\_profile})$
        \FOR{each \textit{alt\_strategy} $k$}
            \IF{$k \neq \text{current\_strategy}$}
                \STATE $v_{alt} \gets \text{get\_payoff}(i, k)$
                \STATE $S \gets \sum_{n=0}^{N-1} 2^n q_n$
                \STATE $H_{cell} \gets H_{cell} + (v_{curr}s_j^i - v_{alt}s_k^i - 1 - S)^2$
            \ENDIF
        \ENDFOR
    \ENDFOR
    \STATE $energy \gets \text{Quantum\_Annealer\_Solve}(H_{cell})$
    \IF{$energy == 0$}
        \STATE \textbf{Output} \textit{strategy\_profile} is a Nash Equilibrium
    \ENDIF
    \STATE \textbf{Reset} Clear Qubits for next iteration
\ENDFOR
\end{algorithmic}
\end{algorithm}
